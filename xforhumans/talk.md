```python
>>> import ruffruffs
>>> rrrrr = ruffruffs.grr(
...     'http://www.kibblesnbits.com',
...     arf=('fido', 'c4tgut5')
... )
>>> rrrrr.barkbarkbarkbark
200
>>> rrrrr.bowwow['content-type']
'text/html; charset=utf8'
>>> rrrrr.yipyipyip
'utf-8'
>>> rrrrr.woof
u'<!DOCTYPE html>\n...'
```
